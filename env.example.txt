# ING Bank LDAP Sunucu Bilgileri
# Bu dosyayı .env olarak kopyalayın ve gerçek değerlerle güncelleyin

# LDAP Sunucu Bilgileri
LDAP_SERVER=ldaps://bankanet.com.tr:636
LDAP_SSL_CERTIFICATE=/data/starburst/SSL/ldap.crt
LDAP_SSL_VERIFY=true
LDAP_ALLOW_INSECURE=false

# LDAP DN Bilgileri
LDAP_BASE_DN=DC=domain,DC=bankanet,DC=com,DC=tr
LDAP_USER_BASE_DN=OU=IngBankUsers,DC=domain,DC=bankanet,DC=com,DC=tr

# Admin Kullanıcı Bilgileri
LDAP_BIND_DN=CN=SVCDATABEES,OU=Users,OU=Applications,DC=domain,DC=bankanet,DC=com,DC=tr
LDAP_BIND_PASSWORD=your_secure_password_here

# Grup Kontrolü
LDAP_GROUP_DN=CN=StarburstUsers,OU=INGBank Security Groups,OU=IngBankUsers,DC=domain,DC=bankanet,DC=com,DC=tr
LDAP_GROUP_AUTH_PATTERN=(&(sAMAccountName=${USER})(memberOf=CN=StarburstUsers,OU=INGBank Security Groups,OU=IngBankUsers,DC=domain,DC=bankanet,DC=com,DC=tr))

# LDAP Attribute'ları
LDAP_USER_FILTER_ATTR=sAMAccountName
LDAP_GROUP_MEMBER_ATTR=memberOf

# Uygulama Ayarları
STREAMLIT_SERVER_PORT=8501
STREAMLIT_SERVER_ADDRESS=0.0.0.0
STREAMLIT_LOGGER_LEVEL=info

# Güvenlik Notları:
# 1. Bu dosyayı .env olarak kopyalayın: cp env.example.txt .env
# 2. Gerçek şifreleri girin
# 3. SSL sertifika yolunu kontrol edin
# 4. .env dosyasını git'e commit etmeyin
# 5. .gitignore'a .env ekleyin
